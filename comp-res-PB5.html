# For k/N lotteries where k in {5,6}

# Order Statistic Probabilities:
# P(Y_(i) = r) = choose(r-1, i-1) * choose(N-r, k-i) / choose(N, k)
order_prob <- function(N,k,i) {
x = seq(1:N)
y = choose(x-1,i-1) * choose(N-x,k-i) / choose(N,k)
return(y)
}

# Order Statistic Expected Values:
# E[Y_(i)] = (N+1)i/(k+1)
order_exp_vec <- function(N,k) {
  x = c()
  for(i in 1:k){
    y = i*(N+1)/(k+1)
    x = c(x,y)
  }
  return(round(x,3))
}


# Order Statistic Covariance:
order_exp_cov <- function(N,k,i,j) {
  y = i*(k-j+1)*(N+1)*(N-k)/((k+2)*(k+1)^2)
  return(y)
}

# Order Statistic Covariance Matrix:
order_exp_mtx <- function(N,k) {
  x = c()
  for(i in 1:k){
    for(j in 1:k){
      y = order_exp_cov(N,k,min(i,j),max(i,j))
      x = c(x,y)
    }
  }
  V = matrix(
    x,
    nrow = k,
    ncol = k,
    byrow = TRUE
  )
  return(V)
}

# Order statistic histograms of historical results with probabilities
# Make sure data frame df has column results in order 1:k
order_hist <- function(N,k,df,lotto_name,period) {
  x = seq(1:N)
  plot_labs = c()
  for(i in 1:k){
    plot_lab = paste(lotto_name," Observed Y_(",i,"), ",period,", with theoretical density", sep="")
    plot_labs = c(plot_labs, plot_lab)
  y = order_prob(N,k,i)
  xy = data.frame(x,y)
  hist_plot = ggplot() + 
    geom_histogram(data=df, aes(x=df[,i+1], y = ..density..), binwidth = 1, color="black", fill="grey") +
    geom_line(xy, mapping=aes(x=x,y=y),color="red") + 
    labs(x = plot_labs[i])
  print(hist_plot)
}}

# Generate stats and vectors
df_stats <- function(N,k,df){

  m.df = nrow(df)
  
  mean_vec = c(mean(df$N1),mean(df$N2),mean(df$N3),mean(df$N4),mean(df$N5))
  if(k==6){
    mean_vec = c(mean_vec, mean(df$N6))
  }
  
  y.bar = matrix(
    mean_vec,
    nrow = k,
    ncol = 1,
    byrow = TRUE
    )

  mu = order_exp_vec(N,k)
  x = y.bar - mu

  V.inv = solve(order_exp_mtx(N,k))
  
  Q = m.df*t(x)%*%V.inv%*%x
  
  p = pchisq(Q,k,lower.tail=FALSE)[1]
  
  return_list <- list("m" = m.df, "y.bar" = round(y.bar[,1],3), "mu" = round(mu,3), "Q" = round(Q,3), "p" = round(p,3), "V.inv" = V.inv)
  return(return_list)
}

# Calculate p-value from Monte Carlo simulation, plot simulated Q distribution
monte_carlo_stats <- function(df,Q,lotto_name,m){
  cat("Observed historical Q = ",round(Q,3),"\n")
  cat("Min Q value from Monte Carlo = ",round(min(df$Q),3),"\n")
  cat("Max Q value from Monte Carlo = ", round(max(df$Q),3),"\n")
  cat("Monte Carlo p-value = ", round(length(which(df$Q > Q[1])) / nrow(df),3))
  x_lab = paste(lotto_name, " Monte Carlo Q, ",nrow(df), " repetitions of ", m, " draws\n (observed Q marked)", sep="")
  MC.plot = ggplot() + 
    geom_histogram(data=df, aes(x=Q, y = ..density..), binwidth = 1, color="black", fill="grey") +
    geom_vline(xintercept = Q, color="red", linetype = "dashed") + 
    labs(x = x_lab)
  print(MC.plot)
}
